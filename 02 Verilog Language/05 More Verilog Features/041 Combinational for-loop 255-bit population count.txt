module top_module( 
    input [254:0] in,
    output [7:0] out );
    
    integer i, n;
    
    always @(*) begin
        n = 0;
        for (i = 0; i < 255; i = i + 1) begin
            if (in[i] == 1) begin
                n = n + 1;
            end
        end
    end
    
    assign out = n;

endmodule
