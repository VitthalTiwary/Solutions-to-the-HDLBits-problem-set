module top_module( 
    input [99:0] in,
    output out_and,
    output out_or,
    output out_xor 
);
    
    integer i;
    
    always @(*) begin
    out_and = 1;
    out_or = 0;
    out_xor = 0;
        for (i = 0; i < 100; i = i + 1) begin
            out_and = out_and & in[i];
            out_or = out_or | in[i];
            out_xor = out_xor ^ in[i];
        end
    end

endmodule
